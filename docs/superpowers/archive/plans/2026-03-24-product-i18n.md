# Product Data i18n Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Convert hardcoded English strings in product catalog, 5 market spec files, and bending-machines equipment data to bilingual (en/zh) translations using slug-based key derivation.

**Architecture:** Data files stay unchanged. Components derive translation keys from slugs at render time. Two namespaces: `catalog.*` in `critical.json` (product pages) and `capabilities.equipment.*` in `deferred.json` (bending machines). Shared lookups for technical labels, column headers, and row cell values.

**Tech Stack:** Next.js 16, next-intl 4.8, TypeScript 5.9

**Spec:** `docs/superpowers/archive/specs/2026-03-24-product-i18n-design.md`

// src/lib/i18n/product-keys.ts

/** Convert column header display string to camelCase translation key.
 *  "Wall Thickness" -> "wallThickness" */
export function columnToKey(col: string): string {
  return col
    .replace(/\s+(.)/g, (_, c: string) => c.toUpperCase())
    .replace(/^(.)/, (c) => c.toLowerCase());
}

/** Convert row cell display string to camelCase translation key.
 *  "Bell End" -> "bellEnd", "Push-fit" -> "pushFit" */
export function cellToKey(cell: string): string {
  return cell
    .replace(/[-/]/g, " ")
    .replace(/\s+(.)/g, (_, c: string) => c.toUpperCase())
    .replace(/^(.)/, (c) => c.toLowerCase());
}
```

// src/constants/product-specs/__tests__/i18n-parity.test.ts
import { describe, expect, it } from "vitest";
import { PRODUCT_CATALOG } from "@/constants/product-catalog";
import { NORTH_AMERICA_SPECS } from "@/constants/product-specs/north-america";
import { AUSTRALIA_NZ_SPECS } from "@/constants/product-specs/australia-new-zealand";
import { MEXICO_SPECS } from "@/constants/product-specs/mexico";
import { EUROPE_SPECS } from "@/constants/product-specs/europe";
import { PNEUMATIC_SPECS } from "@/constants/product-specs/pneumatic-tube-systems";
import { EQUIPMENT_SPECS } from "@/constants/equipment-specs";
import type { MarketSpecs } from "@/constants/product-specs/types";
import enCritical from "../../../../messages/en/critical.json";
import zhCritical from "../../../../messages/zh/critical.json";
import enDeferred from "../../../../messages/en/deferred.json";
import zhDeferred from "../../../../messages/zh/deferred.json";

const SPECS_BY_MARKET: Record<string, MarketSpecs> = {
  "north-america": NORTH_AMERICA_SPECS,
  "australia-new-zealand": AUSTRALIA_NZ_SPECS,
  mexico: MEXICO_SPECS,
  europe: EUROPE_SPECS,
  "pneumatic-tube-systems": PNEUMATIC_SPECS,
};

function getNestedValue(obj: Record<string, unknown>, path: string): unknown {
  return path.split(".").reduce<unknown>(
    (acc, key) => (acc && typeof acc === "object" ? (acc as Record<string, unknown>)[key] : undefined),
    obj,
  );
}

describe("Product i18n parity", () => {
  it("every market has label and description keys in both locales", () => {
    for (const market of PRODUCT_CATALOG.markets) {
      const basePath = `catalog.markets.${market.slug}`;
      expect(getNestedValue(enCritical, `${basePath}.label`), `missing en: ${basePath}.label`).toBeDefined();
      expect(getNestedValue(zhCritical, `${basePath}.label`), `missing zh: ${basePath}.label`).toBeDefined();
      expect(getNestedValue(enCritical, `${basePath}.description`), `missing en: ${basePath}.description`).toBeDefined();
      expect(getNestedValue(zhCritical, `${basePath}.description`), `missing zh: ${basePath}.description`).toBeDefined();
    }
  });

  it("every family has label and description keys in both locales", () => {
    for (const family of PRODUCT_CATALOG.families) {
      const basePath = `catalog.families.${family.marketSlug}.${family.slug}`;
      expect(getNestedValue(enCritical, `${basePath}.label`), `missing en: ${basePath}.label`).toBeDefined();
      expect(getNestedValue(zhCritical, `${basePath}.label`), `missing zh: ${basePath}.label`).toBeDefined();
    }
  });

  it("every technical property key has a shared label translation", () => {
    const allKeys = new Set<string>();
    for (const specs of Object.values(SPECS_BY_MARKET)) {
      for (const key of Object.keys(specs.technical)) {
        allKeys.add(key);
      }
    }
    for (const key of allKeys) {
      expect(getNestedValue(enCritical, `catalog.technicalLabels.${key}`), `missing technicalLabel: ${key}`).toBeDefined();
    }
  });

  it("every family highlight count matches translation keys", () => {
    for (const [marketSlug, specs] of Object.entries(SPECS_BY_MARKET)) {
      for (const family of specs.families) {
        for (let i = 0; i < family.highlights.length; i++) {
          const path = `catalog.specs.${marketSlug}.families.${family.slug}.highlights.${i}`;
          expect(getNestedValue(enCritical, path), `missing: ${path}`).toBeDefined();
          expect(getNestedValue(zhCritical, path), `missing zh: ${path}`).toBeDefined();
        }
      }
    }
  });

  it("every equipment spec has name, param labels, and highlights in both locales", () => {
    for (const equipment of EQUIPMENT_SPECS) {
      const base = `capabilities.equipment.${equipment.slug}`;
      expect(getNestedValue(enDeferred, `${base}.name`), `missing en: ${base}.name`).toBeDefined();
      expect(getNestedValue(zhDeferred, `${base}.name`), `missing zh: ${base}.name`).toBeDefined();
      for (const key of Object.keys(equipment.params)) {
        expect(getNestedValue(enDeferred, `${base}.params.${key}`), `missing: ${base}.params.${key}`).toBeDefined();
      }
      for (let i = 0; i < equipment.highlights.length; i++) {
        expect(getNestedValue(enDeferred, `${base}.highlights.${i}`), `missing: ${base}.highlights.${i}`).toBeDefined();
      }
    }
  });
});
```
